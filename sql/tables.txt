CREATE DATABASE db_collection_music

CREATE TABLE `tb_users` ( `cod_usr` int(11) NOT NULL AUTO_INCREMENT, `usr` varchar(70) NOT NULL, `PASSWORD` varchar(100) DEFAULT NULL, PRIMARY KEY (`cod_usr`) ) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8 

CREATE TABLE `tb_artists` ( `cod_artist` int(11) NOT NULL AUTO_INCREMENT, `artist` varchar(100) NOT NULL, `twitter` varchar(50) NOT NULL, `about` varchar(100) NOT NULL, `deactivate` tinyint(1) DEFAULT NULL, `usr` int(11) NOT NULL, PRIMARY KEY (`cod_artist`), KEY `usr` (`usr`), CONSTRAINT `usr` FOREIGN KEY (`usr`) REFERENCES `tb_users` (`cod_usr`) ) ENGINE=InnoDB AUTO_INCREMENT=18 DEFAULT CHARSET=utf8


CREATE TABLE `tb_musics` ( `cod_music` int(11) NOT NULL AUTO_INCREMENT, `music` varchar(70) NOT NULL, `artist_msc` int(11) NOT NULL, `album` int(11) NOT NULL, `usr_msc` int(11) DEFAULT NULL, `path` varchar(120) NOT NULL, `deactive` tinyint(1) NOT NULL, `date_create` date NOT NULL, PRIMARY KEY (`cod_music`), KEY `album` (`album`), KEY `artist_msc` (`artist_msc`), KEY `usr_msc` (`usr_msc`), CONSTRAINT `album` FOREIGN KEY (`album`) REFERENCES `tb_albums` (`cod_album`), CONSTRAINT `artist_msc` FOREIGN KEY (`artist_msc`) REFERENCES `tb_artists` (`cod_artist`), CONSTRAINT `usr_msc` FOREIGN KEY (`usr_msc`) REFERENCES `tb_users` (`cod_usr`) ) ENGINE=InnoDB AUTO_INCREMENT=5 DEFAULT CHARSET=utf8


CREATE TABLE `tb_albums` ( `cod_album` int(11) NOT NULL AUTO_INCREMENT, `album` varchar(70) NOT NULL, `artist` int(11) NOT NULL, `usr_album` int(11) DEFAULT NULL, `img_cover` varchar(120) DEFAULT NULL, `year_album` int(4) NOT NULL, PRIMARY KEY (`cod_album`), KEY `artist` (`artist`), KEY `usr_album` (`usr_album`), CONSTRAINT `artist` FOREIGN KEY (`artist`) REFERENCES `tb_artists` (`cod_artist`), CONSTRAINT `usr_album` FOREIGN KEY (`usr_album`) REFERENCES `tb_users` (`cod_usr`) ) ENGINE=InnoDB AUTO_INCREMENT=8 DEFAULT CHARSET=utf8
